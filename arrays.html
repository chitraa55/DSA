            void printArray(int arr[], int n) {
                int i;
                for ( i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                printf("\n");
            }
            
            void generateCombinations(int arr[], int data[], int start, int end, int index, int r) {
                int i;
                if (index == r) {
                printArray(data, r);
                return;
                }
            
                for (i = start; i <= end && end - i + 1 >= r - index; i++) {
                data[index] = arr[i];
                generateCombinations(arr, data, i + 1, end, index + 1, r);
                }
            }
