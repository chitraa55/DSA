5. ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM BINARY SEARCH ALGORITHM THROUGH ITERATION. 


    #include
        #include
        int binary(int ar[],int item,int beg,int end)
        {
        int mid;
        while(beg<=end)
        {
        mid=(beg+end)/2;
        if(ar[mid]==item)
        {
        return mid;
        }
        if(ar[mid]>item)
        {
        end=mid-1;
        }
        else
        {
        beg=mid+1;
        }
        }
        return -1;
        }
