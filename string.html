            while (str[length] != '\0') {
                length++;
            }
            return length;
        }
        
        
        void swap(char str[], int i, int j) {
            char temp = str[i];
            str[i] = str[j];
            str[j] = temp;
        }
        
        
        void permuteAndPrint(char str[], int start, int end) {
            int i;
            if (start == end) {
                printf("%s\n", str);
            } else {
                for ( i = start; i <= end; i++) {
                    swap(str, start, i);
                    permuteAndPrint(str, start + 1, end);
                    swap(str,start,i);
                }
            }
        }
        int main() {
            char str[10];
            int length; 
            printf("Enter a string: ");
            scanf("%s", str);
            length = stringLength(str);
            printf("All possible permutations of the string:\n");
            permuteAndPrint(str, 0, length - 1);
        }

                printf("%s\n", text);
